
REPORT  Z06PROGRAM03.

DATA :      STR1 TYPE STRING,
            STR2 TYPE STRING,
            STR3 TYPE STRING,
            STR4 TYPE STRING,
            LEN TYPE I,
            STR6 TYPE I.

DATA C TYPE I.

SELECTION-SCREEN BEGIN OF BLOCK BK1 WITH FRAME TITLE T1.

SELECTION-SCREEN BEGIN OF LINE.
SELECTION-SCREEN COMMENT 8(20) LB1.
PARAMETER X TYPE I .
SELECTION-SCREEN END OF LINE.

SELECTION-SCREEN BEGIN OF LINE.
SELECTION-SCREEN COMMENT 8(20) LB2.
PARAMETER Y TYPE I DEFAULT 20.
SELECTION-SCREEN END OF LINE.

SELECTION-SCREEN END OF BLOCK BK1.


SELECTION-SCREEN BEGIN OF BLOCK BK2 WITH FRAME TITLE T2.

PARAMETER : ADD RADIOBUTTON GROUP GRP1 USER-COMMAND FC1 ,
            SUB RADIOBUTTON GROUP GRP1,
            MUL RADIOBUTTON GROUP GRP1,
            DIV RADIOBUTTON GROUP GRP1,
            NONE RADIOBUTTON GROUP GRP1 DEFAULT 'X'.

SELECTION-SCREEN END OF BLOCK BK2.

INITIALIZATION.
LB1 = 'ENTER FIRST NUMBER '.
LB2 = 'ENTER SECOND NUMBER '.
T1 = 'INPUT FIELD '.
T2 = ' ARITHEMATIC OPERATIONS'.

AT SELECTION-SCREEN ON RADIOBUTTON GROUP GRP1.
CASE SY-UCOMM.
WHEN 'FC1'.

IF ADD = 'X'.
C = X + Y.
STR1 = 'THE SUM OF TWO NUMBER IS '.
STR2 = C .
CLEAR STR3 .
CONCATENATE STR1 STR2 INTO STR3 SEPARATED BY SPACE .
MESSAGE STR3 TYPE 'I' .

ELSEIF SUB = 'X'.
C = X - Y.
*IF STR3 >= 0.
IF STRLEN( STR3 ) >= 0.
STR1 = 'THE SUBTRACTION OF TWO NUMBR IS '.
STR2 = C.
CLEAR STR3.
CONCATENATE STR1 STR2 INTO STR3 SEPARATED BY SPACE .

ELSE .
STR1 = 'THE DIFFERENCE OF TWO NUMBER IS -'.
STR2 = C.
CLEAR STR4.
CONCATENATE STR1 STR2 INTO STR4 SEPARATED BY SPACE.
LEN = STRLEN( STR4 ).
STR6 = LEN - 1 .
CLEAR STR3.
STR3 = STR4+0(STR6).
ENDIF.
MESSAGE STR3 TYPE 'I'.

ELSEIF MUL = 'X'.
C = X * Y.
STR1 = 'THE PRODUCT OF TWO NUMBER '.
STR2 = C.
CLEAR STR3.
CONCATENATE STR1 STR2 INTO STR3 SEPARATED BY SPACE.
MESSAGE STR3 TYPE  'I'.

ELSEIF DIV = 'X'.
C = X / Y.
STR1 = 'THE DIVISION OF TWO NUMBERS IS '.
STR2 =  C.
CLEAR STR3.
CONCATENATE STR1 STR2 INTO STR3 SEPARATED BY SPACE.
MESSAGE STR3 TYPE  'I'.

ELSE.
  WRITE :/ 'NO RADIO BUTTON IS SELECTED'.
  MESSAGE 'No radio button is selected' TYPE 'I'.



ENDIF.
ENDCASE.
